 55     /**
 56      * Add CSS, construct parent
 57      * @param string $label
 58      * @param mixed $options
 59      */
 60     public function __construct( $label= '', $options = array() ) {
 61         parent::__construct( $label, $options );
 62 
 63         if ( !empty( $this->options ) ) {
 64             $this->add_options( $this->options );
 65         }
 66         
 67         // Add the options CSS
 68         fm_add_style( 'fm_options_css', 'css/fieldmanager-options.css' );
 69         
 70         // Sanitization
 71         $this->sanitize = function( $value ) {
 72             if ( isset( $value ) && is_array( $value ) && !empty( $value ) ) {
 73                 return array_map( 'sanitize_text_field', $value );
 74             } else {
 75                 return sanitize_text_field( $value );
 76             }
 77         };

 80     /**
 81      * Add options
 82      * @param array $options
 83      * @return void
 84      */
 85     public function add_options( $options ) {
 86         $values = array_values( $options );
 87         if ( is_array( $values[0] ) ) {
 88             foreach ( $options as $group => $data ) {
 89                 foreach ( $data as $value => $label ) {
 90                     $this->add_option_data( $value, $label, $group, $group );
 91                 }
 92             }
 93         } else {
 94             $keys = array_keys( $options );
 95             $use_name_as_value = ( array_keys( $keys ) === $keys );
 96             foreach ( $options as $k => $v ) {
 97                 $this->add_option_data( $v, ( $use_name_as_value ? $v : $k ) );
 98             }
 99         }
100     }

102     /**
103      * Generate form elements.
104      * @param mixed $value
105      * @return string HTML
106      */
107     public function form_data_elements( $value ) {
108     
109         if ( !$this->has_built_data ) {
110             if ( $this->datasource ) {
111                 $this->add_options( $this->datasource->get_items() );
112             }
113         
114             // Add the first element to the data array. This is useful for database-based data sets that require a first element.
115             if ( !empty( $this->first_element ) ) array_unshift( $this->data, $this->first_element );
116             $this->has_built_data = True;
117         }
118         
119         // If the value is not in an array, put it in one since sometimes there will be multiple selects
120         if ( !is_array( $value ) && isset( $value ) ) {
121             $value = array( $value );
122         }
123     
124         // Output the data for the form. Child classes will handle the output format appropriate for them.
125         $form_data_elements_html = '';
126 
127         if ( !empty( $this->data ) ) {
128         
129             $current_group = '';
130 
131             foreach( $this->data as $data_element ) {
132             
133                 // If grouped display is desired, check where to add the start and end points
134                 // Note we are assuming the data has come pre-sorted into groups
135                 if( $this->grouped && ( $current_group != $data_element['group'] ) ) {
136                     
137                     // Append the end for the previous group unless this is the first group 
138                     if ( $current_group != '' ) $form_data_elements_html .= $this->form_data_end_group();
139                     
140                     // Append the start of the group
141                     $form_data_elements_html .= $this->form_data_start_group( $data_element['group'] );
142                     
143                     // Set the current group
144                     $current_group = $data_element['group'];
145                 }
146                 
147                 // Get the current element
148                 $form_data_elements_html .= $this->form_data_element( $data_element, $value );
149             }
150             
151             // If this was grouped display, close the final group
152             if( $this->grouped || $this->datasource->grouped ) $form_data_elements_html .= $this->form_data_end_group();
153         }
154 
155         return $form_data_elements_html;
156     
157     }

166      * Helper for output functions to toggle a selected options

168      * @param array $options all valid options

170      * @return string $attribute on match, empty on failure.

172     public function option_selected( $current_option, $options, $attribute ) {
173         if ( ( $options != null && !empty( $options ) ) && in_array( $current_option, $options ) ) return $attribute;

177     /**
178      * Presave function, which handles sanitization and validation
179      * @param mixed $value If a single field expects to manage an array, it must override presave()
180      * @return sanitized values. 
181      */
182     public function presave( $value, $current_value = array() ) {
183         if ( !empty( $this->datasource ) ) {
184             return $this->datasource->presave( $this, $value, $current_value );
185         }
186         foreach ( $this->validate as $func ) {
187             if ( !call_user_func( $func, $value ) ) {
188                 $this->_failed_validation( sprintf(
189                     __( 'Input "%1$s" is not valid for field "%2$s" ' ),
190                     (string) $value,
191                     $this->label
192                 ) );
193             }
194         }
195         return call_user_func( $this->sanitize, $value );
196     }

236         
237         $this->data[] = $data;

  6 /**
  7  * Base class containing core functionality for Fieldmanager
  8  *
  9  * Security features of this class for verifying access and saving data:
 10  * 1. In Fieldmanager_Field::save_fields_for_post(), verify that we are using an assigned post type
 11  *    This is the starting point for $_POST requests to post edit pages.
 12  * 2. In Fieldmanager_Field::save_fields_for_post(), verify that the user can save the post type
 13  * 3. In Fieldmanager_Field::save_fields_for_post(), perform nonce validation
 14  * 4. In Fieldmanager_Field::save_to_post_meta(), call Fieldmanager_Field::presave_all()
 15  *    This is the starting point for save routines that use Fieldmanager as an API
 16  * 5. In Fieldmanager_Field::presave_all(), verify that the number of elements is less than $limit,
 17  *    if $limit is not 1 or infinite (0), and verify that all multi-element arrays are numeric only.
 18  * 6. In Fieldmanager_Field::presave_all(), call Fieldmanager_Field::presave()
 19  * 5. In Fieldmanager_Field::presave(), call all assigned validators; if one returns false, die.
 20  * 6a. For groups (including the root group) in Fieldmanager_Group::presave(), verify that all
 21  *     keys in the submission match the form names of valid children.
 22  * 6b. For all other fields, in Fieldmanager_Field::presave(), sanitize the field to save to the DB.
 23  *     The default sanitizer is sanitize_text_field().
 24  *
 25  * @package Fieldmanager
 26  */

265     /**
266      * Build options into properties and throw errors if developers add an unsupported opt.
267      * @param string $label title of form field
268      * @param array $options with keys matching vars of the field in use.
269      * @throws FM_Developer_Exception if an option is set but not defined in this class or the child class.
270      * @throws FM_Developer_Exception if an option is set but not public.
271      */
272     protected function parse_options( $label, $options ) {
273         if ( is_array( $label ) ) $options = $label;
274         else $options['label'] = $label;
275 
276         foreach ( $options as $k => $v ) {
277             try {
278                 $reflection = new ReflectionProperty( $this, $k ); // Would throw a ReflectionException if item doesn't exist (developer error)
279                 if ( $reflection->isPublic() ) $this->$k = $v;
280                 else throw new FM_Developer_Exception; // If the property isn't public, don't set it (rare)
281             } catch ( Exception $e ) {
282                 $message = sprintf(
283                     __( 'You attempted to set a property <em>%1$s</em> that is nonexistant or invalid for an instance of <em>%2$s</em> named <em>%3$s</em>.' ),
284                     $k, __CLASS__, !empty( $options['name'] ) ? $options['name'] : 'NULL'
285                 );
286                 $title = __( 'Nonexistant or invalid option' );
287                 if ( !self::$debug ) {
288                     wp_die( $message, $title );
289                 } else {
290                     throw new FM_Developer_Exception( $message );
291                 }
292             }
293         }
294     }

296     /**
297      * Generates all markup needed for all form elements in this field.
298      * Could be called directly by a plugin or theme.
299      * @param array $values the current values of this element, in a tree structure if the element has children.
300      * @return string HTML for all form elements.
301      */
302     public function element_markup( $values = array() ) {
303         $values = $this->preload_alter_values( $values );
304         if ( $this->limit == 0 ) {
305             if ( count( $values ) + $this->extra_elements <= $this->starting_count ) {
306                 $max = $this->starting_count;
307             }
308             else {
309                 $max = count( $values ) + $this->extra_elements;
310             }
311         }
312         else {
313             $max = $this->limit;
314         }
315 
316         $classes = array( 'fm-wrapper', 'fm-' . $this->name . '-wrapper' );
317         $fm_wrapper_attrs = array();
318         if ( $this->sortable ) {
319             $classes[] = 'fmjs-sortable';
320         }
321         $classes = array_merge( $classes, $this->get_extra_element_classes() );
322 
323         $out = '';
324 
325         // If this element is part of tabbed output, there needs to be a wrapper to contain the tab content
326         if ( $this->is_tab ) {
327             $tab_display_style = ( $this->parent->child_count > 0 ) ? ' style="display: none"' : '';
328             $out .= '<div id="' . $this->get_element_id() . '-tab" class="wp-tabs-panel"' . $tab_display_style . '>';
329         }
330 
331         // For lists of items where $one_label_per_item = False, the label should go outside the wrapper.
332         if ( !empty( $this->label ) && !$this->one_label_per_item ) {
333             $out .= $this->get_element_label( array( 'fm-label-for-list' ) );
334         }
335 
336         // Find the array position of the "counter" (e.g. in element[0], [0] is the counter, thus the position is 1)
337         $html_array_position = 0; // default is no counter; i.e. if $this->limit = 0
338         if ( $this->limit != 1 ) {
339             $html_array_position = 1; // base situation is formname[0], so the counter is in position 1.
340             if ( $this->parent ) {
341                 $parent = $this->parent;
342                 while ( $parent ) {
343                     $html_array_position++; // one more for having a parent (e.g. parent[this][0])
344                     if ( $parent->limit != 1 ) { // and another for the parent having multiple (e.g. parent[0][this][0])
345                         $html_array_position++;
346                     }
347                     $parent = $parent->parent; // parent's parent; root element has null parent which breaks while loop.
348                 }
349             }
350         }
351 
352         // Checks to see if element has display_if data values, and inserts the data attributes if it does
353         if ( isset( $this->display_if ) && !empty( $this->display_if ) ) {
354             $classes[] = 'display-if';
355             $fm_wrapper_attrs['data-display-src'] = $this->display_if['src'];
356             $fm_wrapper_attrs['data-display-value'] = $this->display_if['value'];
357         }
358         $fm_wrapper_attr_string = '';
359         foreach ( $fm_wrapper_attrs as $attr => $val ) {
360             $fm_wrapper_attr_string .= sprintf( '%s="%s" ', $attr, htmlentities( $val ) );
361         }
362         $out .= sprintf( '<div class="%s" data-fm-array-position="%d" %s>',
363             implode( ' ', $classes ),
364             $html_array_position,
365             $fm_wrapper_attr_string
366         );
367 
368         // After starting the field, apply a filter to allow other plugins to append functionality
369         $out = apply_filters( 'fm_element_markup_start', $out, $this );
370 
371         for ( $i = 0; $i < $max; $i++ ) {
372             $this->seq = $i;
373             if ( $this->limit == 1 ) {
374                 $value = $values;
375             } else {
376                 $value = isset( $values[ $i ] ) ? $values[ $i ] : Null;
377             }
378             $out .= $this->single_element_markup( $value );
379         }
380         if ( $this->limit == 0 ) {
381             $out .= $this->add_another();
382         }
383 
384         // Before closing the field, apply a filter to allow other plugins to append functionality
385         $out = apply_filters( 'fm_element_markup_end', $out, $this );
386 
387         $out .= '</div>';
388 
389         // Close the tab wrapper if one exists
390         if ( $this->is_tab ) $out .= '</div>';
391 
392         return $out;
393     }

395     /**
396      * Generate wrappers and labels for one form element.
397      * Is called by element_markup(), calls form_element().
398      * @see Fieldmanager_Field::element_markup()
399      * @see Fieldmanager_Field::form_element()
400      * @param mixed $value the current value of this element.
401      * @param boolean $is_proto true to generate a prototype element for Javascript.
402      * @return string HTML for a single form element.
403      */
404     public function single_element_markup( $value = Null, $is_proto = False ) {
405         $out = '';
406         $classes = array( 'fm-item', 'fm-' . $this->name );
407 
408         self::$global_seq++;
409 
410         // Drop the fm-group class to hide inner box display if no label is set
411         if ( !( $this->field_class == 'group' && ( !isset( $this->label ) || empty( $this->label ) ) ) ) {
412             $classes[] = 'fm-' . $this->field_class;
413         }
414 
415         if ( $this->get_seq() == 0 && $this->limit == 0 ) {
416             // Generate a prototype element for DOM magic on the frontend.
417             if ( $is_proto ) {
418                 $classes[] = 'fmjs-proto';
419             } else {
420                 $this->is_proto = True;
421                 $out .= $this->single_element_markup( Null, True );
422                 $this->is_proto = False;
423             }
424         }
425 
426         $out .= sprintf( '<div class="%s">', implode( ' ', $classes ) );
427 
428         $label = $this->get_element_label( );
429         $render_label_after = False;
430         // Hide the label if it is empty or if this is a tab since it would duplicate the title from the tab label
431         if ( !empty( $this->label ) && !$this->is_tab && $this->one_label_per_item ) {
432             if ( $this->limit == 0 && $this->one_label_per_item ) {
433                 $out .= $this->wrap_with_multi_tools( $label, array( 'fmjs-removable-label' ) );
434             } elseif ( !$this->label_after_element ) {
435                 $out .= $label;
436             } else {
437                 $render_label_after = True;
438             }
439         }
440 
441         $form_element = $this->form_element( $value );
442 
443         if ( $this->limit == 0 && !$this->one_label_per_item ) {
444             $out .= $this->wrap_with_multi_tools( $form_element );
445         } else {
446             $out .= $form_element;
447         }
448 
449         if ( $render_label_after ) $out .= $label;
450 
451         if ( isset( $this->description ) && !empty( $this->description ) ) {
452             $out .= sprintf( '<div class="fm-item-description">%s</div>', $this->description );
453         }
454 
455         $out .= '</div>';
456         return $out;
457     }

491         $tree = $this->get_form_tree();
492         $name = '';
493         foreach ( $tree as $level => $branch ) {
494             if ( 0 == $level ) {
495                 $name .= $branch->name;
496             } else {

498             }
499             if ( $branch->limit != 1 ) {
500                 $name .= '[' . $branch->get_seq() . ']';
501             }
502         }
503         return $name . $multiple;

535     public function get_element_id( ) {
536         $el = $this;

593     /**
594      * Save a submenu page
595      */
596     public function handle_submenu_save() {
597         if ( !empty( $_POST ) && $_GET['page'] == $this->name && current_user_can( $this->submenu_page_args['capability'] ) ) {
598             // Make sure that our nonce field arrived intact
599             if( !wp_verify_nonce( $_POST['fieldmanager-' . $this->name . '-nonce'], 'fieldmanager-save-' . $this->name ) ) {
600                 $this->_unauthorized_access( 'Nonce validation failed' );
601             }
602             $this->data_id = $this->name;
603             $this->data_type = 'options';
604             $current = get_option( $this->name );
605             $data = $this->presave_all( $_POST[ $this->name ], $current );
606             $data = apply_filters( 'fm_submenu_presave_data', $data, $this );
607             if ( get_option( $this->name ) ) {
608                 update_option( $this->name, $data );
609             } else {
610                 add_option( $this->name, $data, ' ', $this->wp_option_autoload ? 'yes' : 'no' );
611             }
612         }
613     }

669     /**
670      * Takes $_POST data and saves it to, calling save_to_post_meta() once validation is passed
671      * When using Fieldmanager as an API, do not call this function directly, call save_to_post_meta()
672      * @param int $post_id
673      * @return void
674      */
675     public function save_fields_for_post( $post_id ) {
676         // Make sure this field is attached to the post type being saved.
677         if ( !isset( $_POST['post_type'] ) || ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) )
678             return;
679         $use_this_post_type = False;
680         foreach ( $this->content_types as $type ) {
681             if ( $type['content_type'] == $_POST['post_type'] ) {
682                 $use_this_post_type = True;
683                 break;
684             }
685         }
686         if ( !$use_this_post_type ) return;
687         if ( $_POST['action'] == 'inline-save' ) return; // no fieldmanager on quick edit yet
688 
689         // Make sure the current user can save this post
690         if( $_POST['post_type'] == 'post' ) {
691             if( !current_user_can( 'edit_post', $post_id ) ) {
692                 $this->_unauthorized_access( 'User cannot edit this post' );
693                 return;
694             }
695         }
696 
697         // Make sure that our nonce field arrived intact
698         if( !wp_verify_nonce( $_POST['fieldmanager-' . $this->name . '-nonce'], 'fieldmanager-save-' . $this->name ) ) {
699             $this->_unauthorized_access( 'Nonce validation failed' );
700         }
701 
702         $this->save_to_post_meta( $post_id, $_POST[ $this->name ] );
703     }

706      * Helper to save an array of data to post meta
707      * @param int $post_id
708      * @param array $data
709      * @return void
710      */
711     public function save_to_post_meta( $post_id, $data ) {
712         if ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) return;
713         $this->data_id = $post_id;

724     /**
725      * Presaves all elements in what could be a set of them, dispatches to $this->presave()
726      * @input mixed[] $values
727      * @return mixed[] sanitized values
728      */
729     public function presave_all( $values, $current_values ) {
730         if ( $this->limit == 1 ) {
731             $values = $this->presave_alter_values( array( $values ), array( $current_values ) );
732             $value = $this->presave( $values[0], $current_values );
733             if ( $this->save_empty || !empty( $value ) ) {
734                 if ( !empty( $this->index ) ) $this->save_index( array( $value ), array( $current_values ) );
735             }
736             return $value;
737         }
738 
739         // If $this->limit != 1, and $values is not an array, that'd just be wrong, and possibly an attack, so...
740         if ( $this->limit != 1 && !is_array( $values ) ) {
741             $this->_unauthorized_access( '$values should be an array because $limit is ' . $this->limit );
742         }
743 
744         // If $this->limit is not 0 or 1, and $values has more than $limit, that could also be an attack...
745         if ( $this->limit > 1 && count( $values ) > $this->limit ) {
746             $this->_unauthorized_access(
747                 sprintf( 'submitted %1$d values against a limit of %2$d', count( $values ), $this->limit )
748             );
749         }
750 
751         if ( isset( $values['proto'] ) ) {
752             unset( $values['proto'] );
753         }
754 
755         $values = $this->presave_alter_values( $values, $current_values );
756 
757         foreach ( $values as $i => $value ) {
758             if ( !is_numeric( $i ) ) {
759                 // If $this->limit != 1 and $values contains something other than a numeric key...
760                 $this->_unauthorized_access( '$values should be a number-indexed array, but found key ' . $i );
761             }
762             $values[$i] = $this->presave( $value, empty( $current_values[$i] ) ? array() : $current_values[$i] );
763             if ( !$this->save_empty && empty( $values[$i] ) ) unset( $values[$i] );
764         }
765         if ( !empty( $this->index ) ) $this->save_index( $values, $current_values );
766         return $values;
767     }

769     /**
770      * Optionally save fields to a separate postmeta index for easy lookup with WP_Query
771      * @param array $values
772      * @return void
773      */
774     protected function save_index( $values, $current_values ) {
775         if ( $this->data_type != 'post' || empty( $this->data_id ) ) return;
776         // Must delete current values specifically, then add new ones, to support a scenario where the
777         // same field in repeating groups with limit = 1 is going to create more than one entry here, and
778         // if we called update_post_meta() we would overwrite the index with each new group.
779         foreach ( $current_values as $v ) {
780             if ( empty( $v ) ) $v = 0; // false or null should be saved as 0 to prevent duplicates
781             delete_post_meta( $this->data_id, $this->index, $v );
782         }
783         // add new values
784         foreach ( $values as $v ) {
785             if ( empty( $v ) ) $v = 0; // false or null should be saved as 0 to prevent duplicates
786             add_post_meta( $this->data_id, $this->index, $v );
787         }
788     }

799     /**
800      * Presave function, which handles sanitization and validation
801      * @param mixed $value If a single field expects to manage an array, it must override presave()
802      * @return sanitized values.
803      */
804     public function presave( $value, $current_value = array() ) {
805         // It's possible that some elements (Grid is one) would be arrays at
806         // this point, but those elements must override this function. Let's
807         // make sure we're dealing with one value here.
808         if ( is_array( $value ) ) {
809             $this->_unauthorized_access( 'presave() in the base class should not get arrays, but did.' );
810         }
811         foreach ( $this->validate as $func ) {
812             if ( !call_user_func( $func, $value ) ) {
813                 $this->_failed_validation( sprintf(
814                     __( 'Input "%1$s" is not valid for field "%2$s" ' ),
815                     (string) $value,
816                     $this->label
817                 ) );
818             }
819         }
820         return call_user_func( $this->sanitize, $value );
821     }

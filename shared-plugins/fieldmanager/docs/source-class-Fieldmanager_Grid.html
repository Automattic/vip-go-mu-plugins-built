 32     /**
 33      * Constructor which adds several scrips and CSS
 34      * @param string $label
 35      * @param array $options
 36      */
 37     public function __construct( $label = '', $options = array() ) {
 38         $this->attributes = array(
 39             'size' => '50',
 40         );
 41         parent::__construct( $label, $options );
 42         $this->sanitize = function( $row, $col, $values ) {
 43             foreach ( $values as $k => $val ) {
 44                 $values[$k] = sanitize_text_field( $val );
 45             }
 46             return $values;
 47         };
 48 
 49         fm_add_script( 'handsontable', 'js/grid/jquery.handsontable.js' );
 50         fm_add_script( 'contextmenu', 'js/grid/lib/jQuery-contextMenu/jquery.contextMenu.js' );
 51         fm_add_script( 'ui_position', 'js/grid/lib/jQuery-contextMenu/jquery.ui.position.js' );
 52         fm_add_script( 'grid', 'js/grid.js' );
 53         fm_add_style( 'context_menu_css', 'js/grid/lib/jQuery-contextMenu/jquery.contextMenu.css' );
 54         fm_add_style( 'handsontable_css', 'js/grid/jquery.handsontable.css' );
 55     }
 56 
 57     /**
 58      * Render HTML for Grid element
 59      * @param array $value
 60      * @return string
 61      */
 62     public function form_element( $value = '' ) {
 63         $grid_activate_id = 'grid-activate-' . uniqid();
 64         if ( !empty( $value ) && is_callable( $this->grid_sort ) ) {
 65             $value = call_user_func( $this->grid_sort, $value );
 66         }
 67         $out = sprintf(
 68             '<div class="grid-toggle-wrapper">
 69                 <div class="fm-grid" id="%2$s" data-fm-grid-name="%1$s"></div>
 70                 <input name="%1$s" class="fm-element" type="hidden" value="%3$s" />
 71                 <p><a href="#" class="grid-activate" id="%6$s" data-with-grid-title="%5$s">%4$s</a></p>
 72             </div>',
 73             $this->get_form_name(),
 74             'hot-grid-id-' . uniqid(), // handsontable must have an ID, but we don't care what it is.
 75             htmlspecialchars( json_encode( $value ) ),
 76             __( 'Show Data Grid' ),
 77             __( 'Hide Data Grid' ),
 78             $grid_activate_id
 79         );
 80         $out .= sprintf("
 81             <script type=\"text/javascript\">
 82                 jQuery( document ).ready( function() {
 83                     jQuery( '#%s' ).one( 'click', function( e ) {
 84                         e.preventDefault();
 85                         var grid = jQuery( this ).parents( '.grid-toggle-wrapper' ).find( '.fm-grid' )[0];
 86                         jQuery( grid ).fm_grid( %s );
 87                     } );
 88                 } );
 89             </script>",
 90             $grid_activate_id,
 91             json_encode( $this->js_options )
 92         );
 93         return $out;
 94     }

 96     /**
 97      * Override presave, using the sanitize function per cell
 98      * @param string $value
 99      * @return array sanitized row/col matrix
100      */
101     public function presave( $value, $current_value = array() ) {
102         $rows = json_decode( stripslashes( $value ), TRUE );
103         foreach ( $rows as $i => $cells ) {
104             foreach ( $cells as $k => $cell ) {
105                 $cell = call_user_func( $this->sanitize, $i, $k, $cell );
106             }
107         }
108         return $rows;
109     }

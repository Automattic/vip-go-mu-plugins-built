 53     /**
 54      * Get posts which match this datasource, optionally filtered by 
 55      * a fragment, e.g. for Autocomplete.
 56      * @param string $fragment
 57      * @return array post_id => post_title for display or AJAX
 58      */
 59     public function get_items( $fragment = Null ) {
 60         if ( is_callable( $this->query_callback ) ) {
 61             return call_user_func( $this->query_callback, $fragment );
 62         }
 63         $default_args = array(
 64             'numberposts' => 10,
 65             'orderby' => 'post_date',
 66             'order' => 'DESC',
 67             'post_status' => 'publish',
 68             'post_type' => 'any',
 69             'suppress_filters' => False,
 70         );
 71         $post_args = array_merge( $default_args, $this->query_args );
 72         $ret = array();
 73         if ( $fragment ) {
 74             $post_id = $exact_post = Null;
 75             if ( preg_match( '/^https?\:/i', $fragment ) ) {
 76                 $url = esc_url( $fragment );
 77                 $url_parts = parse_url( $url );
 78                 $get_vars = array();
 79                 parse_str( $url_parts['query'], $get_vars );
 80                 if ( !empty( $get_vars['post'] )  ) {
 81                     $post_id = intval( $get_vars['post'] );
 82                 } elseif ( !empty( $get_vars['p'] ) ) {
 83                     $post_id = intval( $get_vars['p'] );
 84                 } else {
 85                     $post_id = fm_url_to_post_id( $fragment );
 86                 }
 87             } elseif ( is_numeric( $fragment ) ) {
 88                 $post_id = intval( $fragment );
 89             }
 90             if ( $post_id ) {
 91                 $exact_post = get_post( $post_id );
 92                 if ( $exact_post && (
 93                     $post_args['post_type'] == 'any' ||
 94                     $post_args['post_type'] == $exact_post->post_type ||
 95                     in_array( $exact_post->post_type, $post_args['post_type'] )
 96                 ) ) {
 97                     $ret[ $post_id ] = html_entity_decode( $exact_post->post_title );
 98                 }
 99             }
100             $this->_fragment = $fragment;
101             add_filter( 'posts_where', array( $this, 'title_like' ), 10, 2 );
102         }
103         $posts = get_posts( $post_args );
104         if ( $fragment ) {
105             remove_filter( 'posts_where', array( $this, 'title_like' ), 10, 2 );
106         }
107         foreach ( $posts as $p ) {
108             $ret[$p->ID] = $p->post_title;
109         }
110         return $ret;
111     }

136      * @param array $values new post values

139     public function presave_alter_values( Fieldmanager_Field $field, $values, $current_values ) {
140         if ( $field->data_type != 'post' || !$this->reciprocal ) return $values;
141         foreach ( $current_values as $reciprocal_post_id ) {
142             delete_post_meta( $reciprocal_post_id, $this->reciprocal, $field->data_id );
143         }

147     /**
148      * Handle reciprocal postmeta
149      * @param int $value
150      * @return string
151      */
152     public function presave( Fieldmanager_Field $field, $value, $current_value ) {
153         if ( empty( $value ) ) return;
154         $value = intval( $value );
155         if( !current_user_can( 'edit_post', $value ) ) {
156             die( 'Tried to refer to post ' . $value . ' which user cannot edit.' );  
157         }
158         if ( $this->reciprocal ) {
159             add_post_meta( $value, $this->reciprocal, $field->data_id );
160         }
161         return $value;
162     }

194 /**
195  * Post URLs to IDs function, supports custom post types.
196  * Borrowed and modified from url_to_postid() in wp-includes/rewrite.php
197  * @author http://betterwp.net/
198  * @param string $url
199  * @return int|boolean post ID on success, false on failure
200  */
201 function fm_url_to_post_id( $url ) {
202     global $wp_rewrite;
203  
204     $url = apply_filters('url_to_postid', $url);
205  
206     // First, check to see if there is a 'p=N' or 'page_id=N' to match against
207     if ( preg_match('#[?&](p|page_id|attachment_id)=(\d+)#', $url, $values ) ) {
208         $id = absint($values[2]);
209         if ( $id )
210             return $id;
211     }
212  
213     // Check to see if we are using rewrite rules
214     $rewrite = $wp_rewrite->wp_rewrite_rules();
215  
216     // Not using rewrite rules, and 'p=N' and 'page_id=N' methods failed, so we're out of options
217     if ( empty( $rewrite ) )
218         return 0;
219  
220     // Get rid of the #anchor
221     $url_split = explode( '#', $url );
222     $url = $url_split[0];
223  
224     // Get rid of URL ?query=string
225     $url_split = explode('?', $url);
226     $url = $url_split[0];
227  
228     // Add 'www.' if it is absent and should be there
229     if ( false !== strpos(home_url(), '://www.') && false === strpos($url, '://www.') )
230         $url = str_replace('://', '://www.', $url);
231  
232     // Strip 'www.' if it is present and shouldn't be
233     if ( false === strpos(home_url(), '://www.') )
234         $url = str_replace('://www.', '://', $url);
235  
236     // Strip 'index.php/' if we're not using path info permalinks
237     if ( !$wp_rewrite->using_index_permalinks() )
238         $url = str_replace('index.php/', '', $url);
239  
240     if ( false !== strpos($url, home_url()) ) {
241         // Chop off http://domain.com
242         $url = str_replace(home_url(), '', $url);
243     } else {
244         // Chop off /path/to/blog
245         $home_path = parse_url(home_url());
246         $home_path = isset( $home_path['path'] ) ? $home_path['path'] : '' ;
247         $url = str_replace($home_path, '', $url);
248     }
249  
250     // Trim leading and lagging slashes
251     $url = trim($url, '/');
252  
253     $request = $url;
254     // Look for matches.
255     $request_match = $request;
256     foreach ( (array)$rewrite as $match => $query) {
257         // If the requesting file is the anchor of the match, prepend it
258         // to the path info.
259         if ( !empty($url) && ($url != $request) && (strpos($match, $url) === 0) )
260             $request_match = $url . '/' . $request;
261  
262         if ( preg_match("!^$match!", $request_match, $matches) ) {
263             // Got a match.
264             // Trim the query of everything up to the '?'.
265             $query = preg_replace("!^.+\?!", '', $query);
266  
267             // Substitute the substring matches into the query.
268             $query = addslashes(WP_MatchesMapRegex::apply($query, $matches));
269  
270             // Filter out non-public query vars
271             global $wp;
272             parse_str($query, $query_vars);
273             $query = array();
274             foreach ( (array) $query_vars as $key => $value ) {
275                 if ( in_array($key, $wp->public_query_vars) )
276                     $query[$key] = $value;
277             }
278  
279         // Taken from class-wp.php
280         foreach ( $GLOBALS['wp_post_types'] as $post_type => $t )
281             if ( $t->query_var )
282                 $post_type_query_vars[$t->query_var] = $post_type;
283  
284         foreach ( $wp->public_query_vars as $wpvar ) {
285             if ( isset( $wp->extra_query_vars[$wpvar] ) )
286                 $query[$wpvar] = $wp->extra_query_vars[$wpvar];
287             elseif ( isset( $_POST[$wpvar] ) )
288                 $query[$wpvar] = $_POST[$wpvar];
289             elseif ( isset( $_GET[$wpvar] ) )
290                 $query[$wpvar] = $_GET[$wpvar];
291             elseif ( isset( $query_vars[$wpvar] ) )
292                 $query[$wpvar] = $query_vars[$wpvar];
293  
294             if ( !empty( $query[$wpvar] ) ) {
295                 if ( ! is_array( $query[$wpvar] ) ) {
296                     $query[$wpvar] = (string) $query[$wpvar];
297                 } else {
298                     foreach ( $query[$wpvar] as $vkey => $v ) {
299                         if ( !is_object( $v ) ) {
300                             $query[$wpvar][$vkey] = (string) $v;
301                         }
302                     }
303                 }
304  
305                 if ( isset($post_type_query_vars[$wpvar] ) ) {
306                     $query['post_type'] = $post_type_query_vars[$wpvar];
307                     $query['name'] = $query[$wpvar];
308                 }
309             }
310         }
311  
312             // Do the query
313             $query = new WP_Query($query);
314             if ( !empty($query->posts) && $query->is_singular )
315                 return $query->post->ID;
316             else
317                 return 0;
318         }
319     }
320     return 0;

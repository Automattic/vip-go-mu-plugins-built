 54 /**
 55  * Get the base URL for this plugin.
 56  * @return string URL pointing to Fieldmanager top directory.

 58 function fieldmanager_get_baseurl() {
 59     $path_override = _fieldmanager_registry( 'baseurl' );
 60     if ( $path_override ) {
 61         return $path_override;

 70 /**
 71  * Wrapper to enqueue_scripts which uses a closure
 72  * @param string $handle
 73  * @param string $path
 74  * @param string[] $deps
 75  * @param boolean $ver
 76  * @param boolean $in_footer
 77  * @param string $data_object
 78  * @param array $data
 79  * @param string $plugin_dir
 80  * @return void
 81  */
 82 function fm_add_script( $handle, $path, $deps = array(), $ver = false, $in_footer = false, $data_object = "", $data = array(), $plugin_dir = "" ) {
 83     if ( !is_admin() ) return;
 84     if ( !$ver ) $ver = FM_GLOBAL_ASSET_VERSION;
 85     if ( $plugin_dir == "" ) $plugin_dir = fieldmanager_get_baseurl(); // allow overrides for child plugins
 86     $add_script = function() use ( $handle, $path, $deps, $ver, $in_footer, $data_object, $data, $plugin_dir ) {
 87         wp_enqueue_script( $handle, $plugin_dir . $path, $deps, $ver );
 88         if ( !empty( $data_object ) && !empty( $data ) ) wp_localize_script( $handle, $data_object, $data );
 89     };
 90     add_action( 'admin_enqueue_scripts', $add_script );
 91 }

 93 /**
 94  * Wrapper to enqueue_style which uses a closure
 95  * @param string $handle
 96  * @param string $path
 97  * @param string[] $deps
 98  * @param boolean $ver
 99  * @param string $media
100  * @return void
101  */
102 function fm_add_style( $handle, $path, $deps = array(), $ver = false, $media = 'all' ) {
103     if( !is_admin() ) return;
104     if ( !$ver ) $ver = FM_GLOBAL_ASSET_VERSION;
105     $add_script = function() use ( $handle, $path, $deps, $ver, $media ) {
106         wp_register_style( $handle, fieldmanager_get_baseurl() . $path, $deps, $ver, $media );
107         wp_enqueue_style( $handle );
108     };
109     add_action( 'admin_enqueue_scripts', $add_script );
110 }

112 /**
113  * A simple static registry for Fieldmanager to keep globals out.
114  * @param string $var what variable to set.
115  * @param mixed $val what value to store for $var.
116  * @return mixed if only $var is set, return stored $val. If both are set, set $val and return void.
117  */
118 function _fieldmanager_registry( $var, $val = NULL ) {
119     static $registry;
120     if ( !is_array( $registry ) ) $registry = array();
121     if ( $val === NULL ) {
122         return isset( $registry[$var] ) ? $registry[$var] : False;
123     }
124     $registry[$var] = $val;
125 }

138 /**
139  * Cheap way to tell if we're looking at a post edit page.
140  * It's a good idea to use this at the beginning of implementations which use metaboxes
141  * to avoid initializing your Fieldmanager chain on every page load.
142  * @return boolean are we editing or creating a post?
143  */
144 function fm_is_post_edit_screen() {
145     return stripos( $_SERVER['PHP_SELF'], '/post.php' ) !== FALSE || stripos( $_SERVER['PHP_SELF'], '/post-new.php' ) !== FALSE;
146 }

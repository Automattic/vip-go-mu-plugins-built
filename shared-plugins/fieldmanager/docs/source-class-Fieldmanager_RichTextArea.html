 43     /**
 44      * Construct default attributes; 50x10 textarea
 45      * @param array $options
 46      * @todo dectect locale and apply correct language file
 47      */
 48     public function __construct( $label, $options = array() ) {
 49 
 50         if ( !self::$has_registered_tinymce ) {
 51             wp_enqueue_script( 'tiny_mce.js', includes_url( 'js/tinymce/tiny_mce.js' ) );
 52             wp_enqueue_script( 'wp-langs-en.js', includes_url( 'js/tinymce/langs/wp-langs-en.js' ) );
 53             self::$has_registered_tinymce = True;
 54             // add tinyMCEPreInit early enough in the process that it's available before TinyMCE inits.
 55             // WP Core may overwrite tinyMCEPreInit, but that's fine with us.
 56             add_action( 'admin_print_scripts', function() {
 57                 printf( '<script>
 58 if ( "undefined" === typeof tinyMCEPreInit ) tinyMCEPreInit = { base: "%s", suffix: "" };
 59 </script>',
 60                     includes_url( 'js/tinymce' )
 61                 );
 62             } );
 63             // Mark the language files as loaded; prevents a baffling JS error.
 64             add_action( 'admin_head', function() {
 65                 printf(
 66                     '<script type="text/javascript">
 67 tinyMCE.ScriptLoader.markDone( "%1$sjs/tinymce/langs/en.js" );
 68 tinyMCE.ScriptLoader.markDone( "%1$sjs/tinymce/themes/advanced/langs/en.js" );
 69 </script>',
 70                     includes_url()
 71                 );
 72             } );
 73         }
 74         $this->attributes = array(
 75             'cols' => '50',
 76             'rows' => '10'
 77         );
 78         $this->sanitize = function( $value ) {
 79             return wp_filter_post_kses( $value );
 80         };
 81         // Unlike WP Core, we init TinyMCE on demand, and preserve its natural ability to move
 82         // about the DOM—richtext.js takes care of initializing our options, which are stored 
 83         // per-field, not globally.
 84         fm_add_script( 'fm_richtext', 'js/richtext.js' );
 85         parent::__construct( $label, $options );
 86     }

 89     /**
 90      * Enqueue JS for TinyMCE control.
 91      * We push out all TinyMCE scripts, because we might have many controls on the page
 92      * and would otherwise have to compare them. Our default editor ships with all these
 93      * controls, anyways.
 94      */
 95     public static function enqueue_scripts() {
 96         wp_enqueue_script( 'wplink' );
 97         wp_enqueue_script( 'wpeditor' );
 98         wp_enqueue_script( 'wpdialogs-popup' );
 99         wp_enqueue_style( 'wp-jquery-ui-dialog' );
100         wp_enqueue_script( 'wp-fullscreen' );
101         add_action( 'admin_print_scripts', function() {
102             $post = get_post(); 
103             $args = array();
104             if ( $post->ID ) {
105                 $args['post'] = $post->ID;
106             }
107             wp_enqueue_media( $args ); // generally on post pages this will not have an impact.
108         } );
109     }

124     /**
125      * Get default TinyMCE options; can be overriden per field type with $this->init_options
126      * @return array of options to pass to TinyMCE in JS.
127      */
128     public function get_mce_options() {
129         $editor_id = $this->get_element_id();
130         $buttons = array(
131             apply_filters( 'mce_buttons', array( 'bold', 'italic', 'strikethrough', 'bullist', 'numlist', 'blockquote', 'justifyleft', 'justifycenter', 'justifyright', 'add_media', 'link', 'unlink', 'wp_more', 'spellchecker', 'fullscreen', 'wp_adv' ), $editor_id ),
132             apply_filters( 'mce_buttons_2', array( 'formatselect', 'underline', 'justifyfull', 'forecolor', 'pastetext', 'pasteword', 'removeformat', 'charmap', 'outdent', 'indent', 'undo', 'redo', 'wp_help', 'code' ), $editor_id ),
133             apply_filters( 'mce_buttons_3', array(), $editor_id ),
134             apply_filters( 'mce_buttons_4', array(), $editor_id ),
135         );
136         $options = array(
137             'mode' => "exact",
138             'theme' => "advanced",
139             'language' => 'en',
140             'skin' => "wp_theme",
141             'editor_css' => "/wp-includes/css/editor.css", 
142             'theme_advanced_toolbar_align' => "left",
143             'theme_advanced_statusbar_location' => "bottom",
144             'theme_advanced_resizing' => true,
145             'theme_advanced_resize_horizontal' => false,
146             'dialog_type' => "modal",
147             'content_css' => apply_filters( 'mce_css', fieldmanager_get_baseurl() . 'css/fieldmanager-richtext-content.css' ),
148             'theme_advanced_toolbar_location' => "top",
149             'theme_advanced_buttons1' => implode( ',', $buttons[0] ),
150             'theme_advanced_buttons2' => implode( ',', $buttons[1] ),
151             'theme_advanced_buttons3' => implode( ',', $buttons[2] ),
152             'theme_advanced_buttons4' => implode( ',', $buttons[3] ),
153             'height' => "250",
154             'width' => "100%",
155         );
156         $options['plugins'] = array( 'inlinepopups', 'spellchecker', 'tabfocus', 'paste', 'media', 'fullscreen', 'wordpress', 'wpeditimage', 'wpgallery', 'wplink', 'wpdialogs' );
157         $options['plugins'] = array_unique( apply_filters('tiny_mce_plugins', $options['plugins'] ) );
158         $options['plugins'] = implode( ',', $options['plugins'] );
159         $options = array_merge( $options, $this->init_options );
160         if ( $this->apply_mce_filters ) {
161             $options['content_css'] = apply_filters( 'mce_css', $options['content_css'] );
162             $options = apply_filters( 'tiny_mce_before_init', $options, $editor_id );
163         }
164         if ( isset( $options['style_formats'] ) && !is_array( $options['style_formats'] ) ) {
165             $options['style_formats'] = json_decode( $options['style_formats'] );
166         }
167  
168         // print_r($options); exit;
169         unset( $options['elements'] );
170         return $options;
171     }

173     /**
174      * Render the form element, which is a textarea by default.
175      * @param mixed $value
176      * @return string HTML
177      */
178     public function form_element( $value = '' ) {
179         if ( !self::$has_added_footer_scripts ) {
180             if ( is_admin() ) {
181                 add_action( 'admin_print_footer_scripts', array( __CLASS__, 'editor_js'), 50 );
182                 add_action( 'admin_footer', array( __CLASS__, 'enqueue_scripts'), 1 );
183             } else {
184                 add_action( 'wp_print_footer_scripts', array( __CLASS__, 'editor_js'), 50 );
185                 add_action( 'wp_footer', array( __CLASS__, 'enqueue_scripts'), 1 );
186             }
187             self::$has_added_footer_scripts = True;
188         }
189         return sprintf(
190             '<textarea class="fm-element fm-richtext" name="%1$s" id="%2$s" %3$s data-mce-options="%5$s" />%4$s</textarea>',
191             $this->get_form_name(),
192             $this->get_element_id(),
193             $this->get_element_attributes(),
194             $value,
195             htmlspecialchars( json_encode( $this->get_mce_options() ), ENT_QUOTES )
196         );
197     }

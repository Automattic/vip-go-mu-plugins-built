 75     public function __construct( $options = array() ) {
 76         global $wp_taxonomies;
 77 
 78         // default to showing empty tags, which generally makes more sense for the types of fields
 79         // that fieldmanager supports
 80         if ( !isset( $options['taxonomy_args']['hide_empty'] ) ) {
 81             $options['taxonomy_args']['hide_empty'] = False;
 82         }
 83 
 84         parent::__construct( $options );
 85         if ( $this->only_save_to_taxonomy ) $this->taxonomy_save_to_terms = True;
 86 
 87         // make post_tag and category sortable via term_order, if they're set as taxonomies, and if
 88         // we're not using Fieldmanager storage
 89         if ( $this->only_save_to_taxonomy && in_array( 'post_tag', $this->get_taxonomies() ) ) {
 90             $wp_taxonomies['post_tag']->sort = True;
 91         }
 92         if ( $this->only_save_to_taxonomy && in_array( 'category', $this->get_taxonomies() ) ) {
 93             $wp_taxonomies['category']->sort = True;
 94         }
 95     }

119     /**
120      * Unique among FM types, the taxonomy datasource can store data outside FM's array.
121      * This is how we add it back into the array for editing.
122      * @param Fieldmanager_Field $field
123      * @param array $values
124      * @return array $values loaded up, if applicable.

126     public function preload_alter_values( Fieldmanager_Field $field, $values ) {
127         if ( $this->only_save_to_taxonomy ) {
128             $taxonomies = $this->get_taxonomies();
129             $terms = wp_get_object_terms( $field->data_id, $taxonomies[0], array( 'orderby' => 'term_order' ) );
130 
131             if ( count( $terms ) > 0 ) {
132                 if ( $field->limit == 1 ) {
133                     return $terms[0]->term_id;
134                 } else {
135                     $ret = array();
136                     foreach ( $terms as $term ) {
137                         $ret[] = $term->term_id;
138                     }
139                     return $ret;
140                 }
141             }
142         }
143         return $values;
144     }

146     /**
147      * Presave hook to set taxonomy data
148      * @param int[] $values
149      * @param int[] $current_values
150      * @return int[] $values
151      */
152     public function presave_alter_values( Fieldmanager_Field $field, $values, $current_values ) {
153         // If this is a taxonomy-based field, must also save the value(s) as an object term
154         if ( $this->taxonomy_save_to_terms && isset( $this->taxonomy ) && !empty( $values ) ) {
155             // Sanitize the value(s)
156             if ( !is_array( $values ) ) {
157                 $values = array( $values );
158             }
159             $tax_values = array();
160             foreach ( $values as &$value ) {
161                 if ( !empty( $value ) ) {
162                     if( is_numeric( $value ) )
163                         $tax_values[] = $value;
164                     else if( is_array( $value ) )
165                         $tax_values = $value;
166                 }
167             }
168             $this->save_taxonomy( $tax_values, $field->data_id );
169         }
170         if ( $this->only_save_to_taxonomy ) return array();
171         return $values;
172     }

181     /**
182      * Save taxonomy data
183      * @param mixed[] $tax_values
184      * @return void
185      */
186     public function save_taxonomy( $tax_values, $data_id ) {
187 
188         $tax_values = array_map( 'intval', $tax_values );
189         $tax_values = array_unique( $tax_values );
190         $taxonomies = $this->get_taxonomies();
191 
192         // Store the each term for this post. Handle grouped fields differently since multiple taxonomies are present.
193         if ( count( $taxonomies ) > 1 ) {
194             // Build the taxonomy insert data
195             $taxonomies_to_save = array();
196             foreach ( $tax_values as $term_id ) {
197                 $term = $this->get_term( $term_id );
198                 if ( empty( $taxonomies_to_save[ $term->taxonomy ] ) ) $taxonomies_to_save[ $term->taxonomy ] = array();
199                 $taxonomies_to_save[ $term->taxonomy ][] = $term_id;
200             }
201             foreach ( $taxonomies_to_save as $taxonomy => $terms ) {
202                 wp_set_object_terms( $data_id, $terms, $taxonomy, $this->append_taxonomy );
203             }
204         } else {
205             wp_set_object_terms( $data_id, $tax_values, $taxonomies[0], $this->append_taxonomy );
206         }
207     }

209     /**
210      * Get taxonomy data per $this->taxonomy_args
211      * @param $value The value(s) currently set for this field
212      * @return array[] data entries for options
213      */
214     public function get_items( $fragment = Null ) {
215 
216         // If taxonomy_hierarchical is set, assemble recursive term list, then bail out.
217         if ( $this->taxonomy_hierarchical ) {
218             $tax_args = $this->taxonomy_args;
219             $tax_args['parent'] = 0;
220             $parent_terms = get_terms( $this->get_taxonomies(), $tax_args );
221             return $this->build_hierarchical_term_data( $parent_terms, $this->taxonomy_args, 0, $fragment );
222         }
223 
224         $tax_args = $this->taxonomy_args;
225         if ( !empty( $fragment ) ) $tax_args['search'] = $fragment;
226         $terms = get_terms( $this->get_taxonomies(), $tax_args );
227 
228         // If the taxonomy list was an array and group display is set, ensure all terms are grouped by taxonomy
229         // Use the order of the taxonomy array list for sorting the groups to make this controllable for developers
230         // Order of the terms within the groups is already controllable via $taxonomy_args
231         // Skip this entirely if there is only one taxonomy even if group display is set as it would be unnecessary
232         if ( count( $this->get_taxonomies() ) > 1 && $this->grouped && $this->allow_optgroups ) {
233             // Group the data
234             $term_groups = array();
235             foreach ( $this->get_taxonomies() as $tax ) {
236                 $term_groups[$tax] = array();
237             }
238             foreach ( $terms as $term ) {
239                 $term_groups[$term->taxonomy][ $term->term_id ] = $term->name;
240             }
241             return $term_groups;
242         }
243 
244         // Put the taxonomy data into the proper data structure to be used for display
245         foreach ( $terms as $term ) {
246             // Store the label for the taxonomy as the group since it will be used for display
247             $key = $this->store_term_taxonomy_id ? $term->term_taxonomy_id : $term->term_id;
248             $stack[ $key ] = $term->name;
249         }
250         return apply_filters( 'fm_datasource_term_get_items', $stack, $terms, $this, $fragment );
251     }

253     /**
254      * Helper to support recursive building of a hierarchical taxonomy list.
255      * @param array $parent_terms
256      * @param array $tax_args as used in top-level get_terms() call.
257      * @param int $depth current recursive depth level.
258      * @param string $fragment optional matching pattern
259      * @return array of terms or false if no children found.
260      */
261     protected function build_hierarchical_term_data( $parent_terms, $tax_args, $depth, $stack = array(), $pattern = '' ) {
262 
263         // Walk through each term passed, add it (at current depth) to the data stack.
264         foreach ( $parent_terms as $term ) {
265             $taxonomy_data = get_taxonomy( $term->taxonomy );
266             $prefix = '';
267 
268             // Prefix term based on depth. For $depth = 0, prefix will remain empty.
269             for ( $i = 0; $i < $depth; $i++ ) {
270                 $prefix .= '--';
271             }
272 
273             $key = $this->store_term_taxonomy_id ? $term->term_taxonomy_id : $term->term_id;
274             $stack[ $key ] = $prefix . ' ' . $term->name;
275 
276             // Find child terms of this. If any, recurse on this function.
277             $tax_args['parent'] = $term->term_id;
278             if ( !empty( $pattern ) ) $tax_args['search'] = $fragment;
279             $child_terms = get_terms( $this->get_taxonomies(), $tax_args );
280             if ( $this->taxonomy_hierarchical_depth == 0 || $depth + 1 < $this->taxonomy_hierarchical_depth ) {
281                 if ( !empty( $child_terms ) ) {
282                     $stack = $this->build_hierarchical_term_data( $child_terms, $this->taxonomy_args, $depth + 1, $stack );
283                 }
284             }
285         }
286         return $stack;
287     }

301     /**
302      * Get term by ID only, potentially using multiple taxonomies
303      * @param int $term_id
304      * @return object|null
305      */
306     private function get_term( $term_id ) {
307         if ( $this->store_term_taxonomy_id ) {
308             global $wpdb;
309             return $wpdb->get_row( $wpdb->prepare( "SELECT t.*, tt.* FROM $wpdb->terms AS t INNER JOIN $wpdb->term_taxonomy AS tt ON t.term_id = tt.term_id WHERE tt.term_taxonomy_id = %d LIMIT 1", $term_id ) );
310         } else {
311             $terms = get_terms( $this->get_taxonomies(), array( 'include' => array( $term_id ), 'number' => 1 ) );
312             return !empty( $terms[0] ) ? $terms[0] : Null;
313         }
314     }

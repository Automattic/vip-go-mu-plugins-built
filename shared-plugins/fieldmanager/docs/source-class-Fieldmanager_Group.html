112     /**
113      * Render the element, iterating over children and calling their form_element() functions.
114      * @param mixed $value
115      */
116     public function form_element( $value = NULL ) {
117         $out = '';
118         $tab_group = '';
119         $tab_group_submenu = '';
120         
121         // We do not need the wrapper class for extra padding if no label is set for the group
122         if ( isset( $this->label ) && !empty( $this->label ) ) $out .= '<div class="fm-group-inner">';
123         
124         // If the display output for this group is set to tabs, build the tab group for navigation
125         if ( $this->tabbed ) $tab_group = sprintf( '<ul class="fm-tab-bar wp-tab-bar" id="%s-tabs">', $this->get_element_id() );
126         
127         // Produce HTML for each of the children
128         foreach ( $this->children as $element ) {
129             
130             $element->parent = $this;
131             
132             // If the display output for this group is set to tabs, add a tab for this child
133             if ( $this->tabbed ) {
134                 
135                 // Set default classes to display the first tab content and hide others
136                 $tab_classes = array( 'fm-tab' );
137                 $tab_classes[] = ( $this->child_count == 0 ) ? "wp-tab-active" : "hide-if-no-js";
138             
139                 // Generate output for the tab. Depends on whether or not there is a tab limit in place.
140                 if ( $this->tab_limit == 0 || $this->child_count < $this->tab_limit ) {
141                     $tab_group .=  sprintf( '<li class="%s"><a href="#%s-tab">%s</a></li>',
142                         implode( " ", $tab_classes ),
143                         $element->get_element_id(),
144                         $element->label
145                      );
146                 } else if ( $this->tab_limit != 0 && $this->child_count >= $this->tab_limit ) {
147                     $submenu_item_classes = array( 'fm-submenu-item' );
148                     $submenu_item_link_class = "";
149 
150                     // Create the More tab when first hitting the tab limit                 
151                     if ( $this->child_count == $this->tab_limit ) {
152                         // Create the tab
153                         $tab_group_submenu .=  sprintf( '<li class="fm-tab fm-has-submenu %s"><a href="#%s-tab">%s</a>',
154                             $tab_class,
155                             $element->get_element_id(),
156                             __( 'More...' )
157                          );
158                          
159                          // Start the submenu
160                          $tab_group_submenu .= sprintf( 
161                             '<div class="fm-submenu" id="%s-submenu"><div class="fm-submenu-wrap fm-submenu-wrap"><ul>',
162                             $this->get_element_id()
163                          );
164                          
165                          // Make sure the first submenu item is designated
166                          $submenu_item_classes[] = 'fm-first-item';
167                          $submenu_item_link_class = 'class="fm-first-item"';
168                     }
169                     
170                     // Add this element to the More menu
171                     $tab_group_submenu .=  sprintf( '<li class="%s"><a href="#%s-tab" %s>%s</a></li>',
172                         implode( ' ', $submenu_item_classes ),
173                         $element->get_element_id(),
174                         $submenu_item_link_class,
175                         $element->label
176                     );
177                 }
178                                 
179                 // Ensure the child is aware it is tab content
180                 $element->is_tab = TRUE;
181             }
182         
183             // Get markup for the child element
184             $child_value = empty( $value[ $element->name ] ) ? Null : $value[ $element->name ];
185 
186             // propagate editor state down the chain
187             if ( $this->data_type ) $element->data_type = $this->data_type;
188             if ( $this->data_id ) $element->data_id = $this->data_id;
189             
190             $out .= $element->element_markup( $child_value );
191             
192             $this->child_count++;
193             
194         }
195         
196         // We do not need the wrapper class for extra padding if no label is set for the group
197         if ( isset( $this->label ) && !empty( $this->label ) ) $out .= '</div>';
198         
199         // If the display output for this group is set to tabs, build the tab group for navigation
200         if ( $this->tab_limit != 0 && $this->child_count >= $this->tab_limit ) $tab_group_submenu .= '</ul></div></div></li>';
201         if ( $this->tabbed ) $tab_group .= $tab_group_submenu . '</ul>';
202 
203         
204         // Return the complete HTML
205         return $tab_group . $out;
206     }

217     /**
218      * Presave override for groups which dispatches to child presave_all methods.
219      * @input mixed[] values
220      * @return mixed[] values
221      */
222     public function presave( $values, $current_values = array() ) {
223         // @SECURITY@ First, make sure all the values we're given are legal.
224         if( isset( $values ) && !empty( $values ) ) {
225             foreach ( array_keys( $values ) as $key ) {
226                 if ( !isset( $this->children[$key] ) ) {
227                     // If we're here, it means that the input, generally $_POST, contains a value that doesn't belong,
228                     // and thus one which we cannot sanitize and must not save. This might be an attack.
229                     $this->_unauthorized_access( sprintf( 'Found "%1$s" in data but not in children', $key ) );
230                 }
231             }
232         }
233         
234         // Then, dispatch them for sanitization to the children.
235         foreach ( $this->children as $k => $element ) {
236             $element->data_id = $this->data_id;
237             $element->data_type = $this->data_type;
238             if ( empty( $values[$element->name] ) ) {
239                 $values[ $element->name ] = NULL;
240             }
241             $child_value = empty( $values[ $element->name ] ) ? Null : $values[ $element->name ];
242             $current_child_value = empty( $current_values[$element->name ]) ? array() : $current_values[$element->name];
243             $values[ $element->name ] = $element->presave_all( $values[ $element->name ], $current_child_value );
244             if ( !$this->save_empty && $this->limit != 1 ) {
245                 if ( is_array( $values[$element->name] ) && empty( $values[$element->name] ) ) unset( $values[$element->name] );
246                 elseif ( empty( $values[$element->name] ) ) unset( $values[$element->name] );
247             }
248         }
249 
250         if ( is_callable( $this->group_is_empty ) ) {
251             if ( call_user_func( $this->group_is_empty, $values ) ) {
252                 $values = array();
253             }
254         }
255 
256         return $values;
257     }

259     /**
260      * Get an HTML label for this element.
261      * @param array $classes extra CSS classes.
262      * @return string HTML label.
263      */
264     public function get_element_label( $classes = array() ) {
265         $classes[] = 'fm-label';
266         $classes[] = 'fm-label-' . $this->name;
267 
268         $wrapper_classes = array( 'fm-group-label-wrapper' );
269 
270         if ( $this->sortable ) {
271             $wrapper_classes[] = 'fmjs-drag';
272             $wrapper_classes[] = 'fmjs-drag-header';
273         }
274 
275         $extra_attrs = '';
276         if ( $this->label_macro ) {
277             $this->label_format = $this->label_macro[0];
278             $this->label_token = sprintf( '.fm-%s input.fm-element', $this->label_macro[1] );
279         }
280 
281         if ( $this->label_format && $this->label_token ) {
282             $extra_attrs = sprintf(
283                 'data-label-format="%1$s" data-label-token="%2$s"',
284                 htmlentities( $this->label_format ),
285                 htmlentities( $this->label_token )
286             );
287             $classes[] = 'fm-label-with-macro';
288         }
289 
290         return sprintf(
291             '<div class="%1$s"><%2$s class="%3$s"%4$s>%5$s</%2$s>%6$s</div>',
292             implode( ' ', $wrapper_classes ),
293             $this->label_element,
294             implode( ' ', $classes ),
295             $extra_attrs,
296             $this->label,
297             $this->limit == 0 ? $this->get_remove_handle() : ''
298         );
299     }
